Example commented code

//The function below updates the current time, to the second.

function update_current_date () {

/* The next line of code declares the variable current_date_string and assigns it a value. CHECK: The value is the Date method and *its* member method toString */

var current_date_string = Date().toString();

/* The variable elem has been assigned the value document (the object), with this object's associated method getElementById with the value "para". i.e. the paragraph element in the HTML document. This will find the 'para' element in the HTML document. */

var elem = document.getElementById("para");

/* The below line of code users the property innerHTML. This replaces the text inside the HTML paragraph tags (para) and assigns the value belonging to current_date_string which, as stated above is Date ().toString() */

elem.innerHTML = current_date_string;
return ;
}

/* The below line of code refers to the object 'window' and it's method setInterval which is given the function outlined above i.e. the output of update_current_date. The interval of this date is then set to 500. */

window.setInterval(update_current_date, 500);

/* The below text declares an array. An array is given a name just like any other variable, in this case it's called classes. Square brackets enclose the assigned values, each of which are separated by a comma. An array stores a list of values that are related to eachother. The technique for creating an array used below is known as an array literal - this is the preferred idiom for creating an array. In this case the values are different colours that we want to use for a flashing coloured background on the webpage. In order to implement this code we create a function called flashing-colour. */

var index = 0;
var classes = ["back_blue", "back_red", "back_green", "back_orange"];
function flashing_colour () {

/* This next line of code refers a className which lies within the body of the document. */

document.body.className = classes [index];

/* Each number in an array is automatically given a number called an index. */

index = index + 1;
if (index >= classes.length) {
index = 0;
} }

/* The window is the object. Here we have used the setInterval method (which belongs to the window object?) and parsed in the function flashing colour, setting the interval to 500 */

window.setInterval(flashing_colour, 500);
